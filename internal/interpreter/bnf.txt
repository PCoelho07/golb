<expression> ::= <directive> <space>* <delimiter_o> <space>+ <backend_list> <space>+ <delimiter_c> <space>
<backend_list> ::= <backend> | <backend> <space>+ <backend_list>
<backend> ::= "http://" <host>
<host> ::= <word> <port_opt>*
<port_opt> ::= ":" <port>
<port> ::= <digit> | <digit> <port>
<digit> ::= [0-9]
<word> ::= <letter_or_digit> | <letter_or_digit> <word>
<letter_or_digit> ::= <letter> | <digit>
<letter> ::= [a-z] | [A-Z]
<directive> ::= "backends"
<delimiter_o> ::= "{"
<delimiter_c> ::= "}"
<space> ::= " " | "\t" | "\n"
